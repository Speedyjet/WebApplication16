CREATE DATABASE LIBRARY
GO 
USE LIBRARY
CREATE TABLE BOOKS(
BOOKID INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
BOOKNAME NVARCHAR(30) NOT NULL UNIQUE,
AUTHOR NVARCHAR(30) NOT NULL UNIQUE,
ISBN INT NOT NULL UNIQUE);
GO
CREATE TABLE CATEGORIES(
CATEGORYID INT NOT NULL PRIMARY KEY IDENTITY(1,1),
CATEGORYNAME NVARCHAR(30) NULL);
GO
CREATE TABLE INTERTABLE(
CATID INT NULL CONSTRAINT FK_CAT FOREIGN KEY (CATID) REFERENCES CATEGORIES(CATEGORYID),
BOOKID INT NOT NULL CONSTRAINT FK_BOOK FOREIGN KEY (BOOKID) REFERENCES BOOKS(BOOKID),
IDENT INT NOT NULL PRIMARY KEY);
